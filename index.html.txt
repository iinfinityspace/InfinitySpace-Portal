    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-app.js";
        import { getAuth, signInAnonymously, signInWithCustomToken, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-auth.js";
        import { getFirestore, collection, addDoc, onSnapshot, doc, deleteDoc, serverTimestamp } from "https://www.gstatic.com/firebasejs/10.8.0/firebase-firestore.js";

        const firebaseConfig = typeof __firebase_config !== 'undefined' ? JSON.parse(__firebase_config) : {
            apiKey: "mock-api-key", authDomain: "mock-auth-domain", projectId: "mock-project", storageBucket: "mock-storage", messagingSenderId: "123", appId: "mock-app"
        };
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'infinity-space';

        const app = initializeApp(firebaseConfig);
        const auth = getAuth(app);
        const db = getFirestore(app);
        let currentUser = null;

        const initAuth = async () => {
            try {
                if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
                    await signInWithCustomToken(auth, __initial_auth_token);
                } else {
                    await signInAnonymously(auth);
                }
            } catch (error) {
                showToast("فشل الاتصال السحابي", "fa-circle-xmark text-red-400");
                const badge = document.getElementById('auth-status-badge');
                if(badge) {
                    badge.className = "px-3 py-1 bg-red-50 border border-red-200 text-red-800 text-xs font-bold rounded-lg flex items-center gap-2";
                    badge.innerText = "فشل الاتصال السحابي";
                }
            }
        };

        initAuth();

        onAuthStateChanged(auth, (user) => {
            currentUser = user;
            if (user) {
                const badge = document.getElementById('auth-status-badge');
                if(badge) {
                    badge.className = "px-3 py-1 bg-emerald-50 border border-emerald-200 text-emerald-700 text-xs font-bold rounded-lg flex items-center gap-2";
                    badge.innerHTML = `<i class="fa-solid fa-cloud-check"></i> متصل كـ: ${user.uid.substring(0, 8)}...`;
                }
                setupRealtimeListener();
            }
        });

        function setupRealtimeListener() {
            if (!currentUser) return;
            const consultationsCol = collection(db, 'artifacts', appId, 'public', 'data', 'consultations');
            
            onSnapshot(consultationsCol, (snapshot) => {
                const requests = [];
                snapshot.forEach((doc) => { requests.push({ id: doc.id, ...doc.data() }); });
                requests.sort((a, b) => {
                    const timeA = a.timestamp?.seconds ? a.timestamp.seconds * 1000 : new Date(a.timestamp).getTime();
                    const timeB = b.timestamp?.seconds ? b.timestamp.seconds * 1000 : new Date(b.timestamp).getTime();
                    return timeB - timeA;
                });
                renderConsultationRows(requests);
            }, (error) => {
                showToast("خطأ في الاستماع للبيانات", "fa-circle-xmark text-red-400");
            });
        }

        function renderConsultationRows(requests) {
            const tbody = document.getElementById('db-requests-tbody');
            if(!tbody) return;
            if (requests.length === 0) {
                tbody.innerHTML = `<tr><td colspan="6" class="p-8 text-center text-slate-400"><i class="fa-solid fa-folder-open text-2xl mb-2 block"></i> لا توجد أي طلبات مسجلة حالياً في قاعدة البيانات السحابية.</td></tr>`;
                return;
            }

            tbody.innerHTML = "";
            requests.forEach(req => {
                const tr = document.createElement('tr');
                tr.className = "hover:bg-slate-50 transition-colors";
                const dateStr = req.timestamp?.seconds ? new Date(req.timestamp.seconds * 1000).toLocaleString('ar-SA') : new Date(req.timestamp || Date.now()).toLocaleString('ar-SA');
                const serviceTexts = { water: "مسح مياه جوفية (ERT)", mining: "استكشاف تعديني وتخريط", geotech: "دراسات تربة وميكانيكا صخور" };

                tr.innerHTML = `
                    <td class="p-4 font-bold text-slate-900">${req.clientName || 'غير معروف'}</td>
                    <td class="p-4">${req.location || 'غير محدد'}</td>
                    <td class="p-4 font-semibold text-cosmic-blueDark">${serviceTexts[req.service] || req.service}</td>
                    <td class="p-4"><span class="px-2 py-0.5 bg-sky-50 text-cosmic-blueDark border border-sky-100 rounded text-[10px] font-bold">نشط</span></td>
                    <td class="p-4 text-slate-400 font-mono text-[10px]">${dateStr}</td>
                    <td class="p-4 flex gap-2">
                        <button class="bg-slate-100 hover:bg-slate-200 text-slate-700 px-2 py-1 rounded text-[10px] font-bold transition-colors export-pdf-btn">تصدير PDF</button>
                        <button class="bg-red-50 hover:bg-red-100 text-red-600 px-2 py-1 rounded text-[10px] font-bold transition-colors delete-btn">حذف</button>
                    </td>
                `;

                tr.querySelector('.export-pdf-btn').addEventListener('click', () => { window.exportSingleProposalPDF(req); });
                tr.querySelector('.delete-btn').addEventListener('click', async () => {
                    try {
                        await deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'consultations', req.id));
                        showToast("تم حذف الطلب بنجاح!", "fa-trash text-emerald-400");
                    } catch (err) { showToast("فشل الحذف", "fa-circle-xmark text-red-400"); }
                });
                tbody.appendChild(tr);
            });
        }

        window.handleFormSubmission = async function(event) {
            event.preventDefault();
            if (!currentUser) { showToast("يجب الاتصال بالسحابة أولاً!", "fa-circle-xmark text-red-400"); return; }

            const btn = document.getElementById('btn-submit-form');
            if(!btn) return;
            const originalHtml = btn.innerHTML;
            btn.disabled = true;
            btn.innerHTML = `<i class="fa-solid fa-circle-notch animate-spin"></i> جاري الحفظ...`;

            const clientName = document.getElementById('prop-client')?.value || '';
            const location = document.getElementById('prop-location')?.value || '';
            const service = document.getElementById('prop-service')?.value || '';

            try {
                const consultationsCol = collection(db, 'artifacts', appId, 'public', 'data', 'consultations');
                await addDoc(consultationsCol, { clientName, location, service, timestamp: serverTimestamp() || new Date().toISOString(), userId: currentUser.uid });
                showToast("تم الحفظ الفوري في Firestore!", "fa-circle-check text-emerald-400");

        function calcPoisson() {
            const vpInp = document.getElementById('poisson-vp');
            const vsInp = document.getElementById('poisson-vs');
            if (!vpInp || !vsInp) return;

            const vp = parseFloat(vpInp.value);
            const vs = parseFloat(vsInp.value);
            appState.poisson.vp = vp; appState.poisson.vs = vs;

            if(document.getElementById('poisson-vp-lbl')) document.getElementById('poisson-vp-lbl').innerText = vp + ' م/ث';
            if(document.getElementById('poisson-vs-lbl')) document.getElementById('poisson-vs-lbl').innerText = vs + ' م/ث';

            const valEl = document.getElementById('poisson-result-val');
            const outValEl = document.getElementById('poi-res-val');
            const interpEl = document.getElementById('poisson-interpretation');
            const outDescEl = document.getElementById('poi-desc');

            if (vs >= vp) {
                if(valEl) valEl.innerText = "خطأ";
                if(outValEl) outValEl.innerText = "خطأ";
                if(interpEl) interpEl.innerText = "Vs يجب أن تكون أقل من Vp";
                return;
            }

            const nu = (vp*vp - 2*vs*vs) / (2*(vp*vp - vs*vs));
            if(valEl) valEl.innerText = nu.toFixed(3);
            if(outValEl) outValEl.innerText = nu.toFixed(3);
            if(document.getElementById('ai-poisson-val')) document.getElementById('ai-poisson-val').innerText = nu.toFixed(3);

            let d = "";
            if (nu < 0.20) {
                d = "صخور هشة مكسرة قابلة للتفتت بسهولة.";
                if(interpEl) interpEl.className = "text-xs font-bold text-red-600 mt-1";
                if(outDescEl) outDescEl.className = "text-sm font-bold text-red-600 bg-red-50 px-3 py-2 rounded-lg inline-block w-full";
            } else if (nu >= 0.20 && nu < 0.30) {
                d = "صخر متماسك ممتاز للإنشاءات العميقة.";
                if(interpEl) interpEl.className = "text-xs font-bold text-emerald-600 mt-1";
                if(outDescEl) outDescEl.className = "text-sm font-bold text-emerald-600 bg-emerald-50 px-3 py-2 rounded-lg inline-block w-full";
            } else {
                d = "طبقة مشبعة بالماء ورواسب سائبة.";
                if(interpEl) interpEl.className = "text-xs font-bold text-amber-600 mt-1";
                if(outDescEl) outDescEl.className = "text-sm font-bold text-amber-600 bg-amber-50 px-3 py-2 rounded-lg inline-block w-full";
            }
            if(interpEl) interpEl.innerText = d;
            if(outDescEl) outDescEl.innerText = d;
        }

        async function fetchGemini(prompt) {
            const apiKey = "";
            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
            const sys = "أنت مهندس جيولوجي ومسوق في شركة Infinity Space.";
            const payload = { contents: [{ parts: [{ text: prompt }] }], systemInstruction: { parts: [{ text: sys }] } };
            
            let d = 1000;
            for(let i=0; i<3; i++) {
                try {
                    const r = await fetch(url, { method: 'POST', headers: {'Content-Type':'application/json'}, body: JSON.stringify(payload) });
                    if(r.ok) { const j = await r.json(); return j.candidates[0].content.parts[0].text; }
                } catch(e) {}
                await new Promise(res => setTimeout(res, d)); d*=2;
            }
            throw new Error("فشل الاتصال بخدمة الذكاء الاصطناعي.");
        }
